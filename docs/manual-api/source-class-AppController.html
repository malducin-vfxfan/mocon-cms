 33: class AppController extends Controller {
 34: /**
 35: * VFXfan CMS system authorization
 36: *
 37: * The base system only uses authentication, meaning a logged in user is by
 38: * default an administrator and can do anything. For better user control
 39: * additonal methods have been implemented: ACL authorization and controller
 40: * based authorizaed. ACL is enabled initially but the component and
 41: * authorization can be easily commented out.
 42: *
 43: * In controller based authentication, an isAuthorized method is used to check
 44: * users permissions. One is implemented here in the App Controller which
 45: * mimcs the default behavior: Administrators can do anything. Controller
 46: * based authorization is useful if only a few groups are needed and only
 47: * a few groups require special authorization.
 48: *
 49: * ACL allows more fine-grained permissions. To use uncomment the lines below
 50: * and also the actsAs variables in the User and Group models. Allow
 51: * access to the system, update the ACOs, which allows Aministrators access
 52: * to the whole system. Then only allow the index and view views. Create
 53: * the AROs and ACOs tables first, schema can be found at:
 54: *
 55: * app/Config/Schema/db_acl.sql
 56: */
 57: /**
 58:  * Components
 59:  *
 60:  * @var array
 61:  */
 62:     public $components = array(
 63:         'Acl',
 64:         'Auth' => array(
 65:             'authenticate' => array(
 66:                 'Form' => array(
 67:                     'passwordHasher' => array(
 68:                         'className' => 'BlowfishAdvanced'
 69:                     )
 70:                 )
 71:             ),
 72:             'authorize' => array(
 73:                 'Actions' => array('actionPath' => 'controllers'), // use ACL based authorization
 74: //              'Controller' // use controller based authorization
 75:             )
 76:         ),
 77:         'Paginator',
 78:         'Security',
 79:         'Session',
 80: //      'DebugKit.Toolbar',
 81:     );
 82: 
 83: /**
 84:  * Helpers
 85:  *
 86:  * @var array
 87:  */
 88:     public $helpers = array('Form', 'Html', 'Menu', 'Session');
 89: 
 90: /**
 91:  * beforeFilter method
 92:  *
 93:  * Set application wide settings, mainly security related ones. By
 94:  * default let views named index and view be always accesible. Also set
 95:  * CSRF duration, a blackhole callback and a session variable to select
 96:  * the default or mobile theme.
 97:  *
 98:  * @return void
 99:  */
100:     public function beforeFilter() {
101:         // always allow index and view access
102:         $this->Auth->allow(array('index', 'view'));
103: 
104:         // set the login and logout actions
105:         $this->Auth->loginAction = array('controller' => 'users', 'action' => 'admin_login');
106:         $this->Auth->loginRedirect = array('controller' => 'posts', 'action' => 'admin_index');
107:         $this->Auth->logoutRedirect = array('controller' => 'users', 'action' => 'admin_login');
108:         $this->Auth->flash['element'] = 'Flash/auth';
109: 
110:         // set security options
111:         $this->Security->csrfExpires = '+10 minutes';
112:         $this->Security->blackHoleCallback = 'blackhole';
113:         $this->Security->requireAuth(array('admin_index', 'admin_view', 'admin_add', 'admin_edit', 'admin_delete'));
114:         Security::setHash('sha256');
115:         Security::setCost(Configure::read('Security.BlowfishAdvanced.cost'));
116: 
117:         // check to see if we are on a mobile device
118:         if (!$this->Session->check('Config.theme')) {
119:             if ($this->request->is('mobile')) {
120:                 $this->Session->write('Config.theme', 'mobile');
121:             } else {
122:                 $this->Session->write('Config.theme', 'default');
123:             }
124:         }
125: 
126:         // use the Mobile theme if configured
127:         if ($this->Session->read('Config.theme') == 'mobile') {
128:             $this->theme = 'Mobile';
129:         } else {
130:             $this->theme = null;
131:         }
132:         $this->response->vary('User-Agent');
133:     }
134: 
135: /**
136:  * isAuthorized method
137:  *
138:  * Generic application wide controller based authorization. By default
139:  * admin action require that the user be logged in and belong to the
140:  * Admin group.
141:  *
142:  * @return boolean
143:  */
144:     public function isAuthorized($user = null) {
145:         // Any registered user can access public functions
146:         if (empty($this->request->params['admin'])) { // could also check $this->request->prefix
147:             return true;
148:         }
149: 
150:         // Only admins can access admin functions
151:         if (isset($this->request->params['admin'])) { // could also check $this->request->prefix
152:             return (bool)($user['Group']['name'] === 'Administrators'); // could also check $this->Auth->user('Group.name')
153:         }
154: 
155:         // Default deny
156:         return false;
157:     }
158: 
159: /**
160:  * blackhole method
161:  *
162:  * Generic application wide blackhole callback. If the problem is
163:  * related to CSRF protection of forms, mainly that the token has
164:  * expired, it redirects to the action again, thus regenarating the
165:  * token. If the problem is Auth related, mainly action mismatches or
166:  * forms that don't validate, it redirects to the action again. All
167:  * other problems throw a Not Found Exception, which shows the 400
168:  * error page that redirects to the main page.
169:  *
170:  * @return void
171:  */
172:     public function blackhole($type) {
173:         if ($type == 'csrf') {
174:             $this->Session->setFlash('The Form has expired, please try again.', 'Flash/error');
175:             return $this->redirect(array('action' => $this->request->action));
176:         }
177:         elseif ($type == 'auth') {
178:             $this->Session->setFlash('There was a problem with the action (probably validation), please try again.', 'Flash/error');
179:             return $this->redirect(array('action' => $this->request->action));
180:         } else {
181:             throw new NotFoundException('Invalid action.');
182:         }
183:     }
