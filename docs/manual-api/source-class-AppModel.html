 44: /**
 45:  * Strips extra whitespace from output
 46:  *
 47:  * @param string $str String to sanitize
 48:  * @return string whitespace sanitized string
 49:  */
 50:     public static function clean($string, $options = array()) {
 51:         static $defaultCharset = false;
 52: 
 53:         if ($defaultCharset === false) {
 54:             $defaultCharset = Configure::read('App.encoding');
 55:             if ($defaultCharset === null) {
 56:                 $defaultCharset = 'UTF-8';
 57:             }
 58:         }
 59:         $default = array(
 60:             'charset' => $defaultCharset,
 61:             'quotes' => ENT_COMPAT | ENT_HTML5,
 62:             'decode_first' => true,
 63:             'remove_chars' => 0,
 64:             'remove' => true,
 65:             'clean_whitespace' => true,
 66:             'encode' => true
 67:         );
 68: 
 69:         $options = array_merge($default, $options);
 70: 
 71:         if ($options['decode_first']) {
 72:             $string = htmlspecialchars_decode($string, ENT_QUOTES | ENT_HTML5);
 73:         }
 74: 
 75:         if (!is_int($options['remove_chars'])) $options['remove_chars'] = 0;
 76:         if ($options['remove_chars'] & self::REMOVE_QUOTES) {
 77:             $string = str_replace(array('"'), '', $string);
 78:         }
 79:         if ($options['remove_chars'] & self::REMOVE_SINGLE_QUOTES) {
 80:             $string = str_replace(array("'"), '', $string);
 81:         }
 82:         if ($options['remove_chars'] & self::REMOVE_AMPERSAND) {
 83:             $string = str_replace(array('&'), '', $string);
 84:         }
 85:         if ($options['remove_chars'] & self::REMOVE_CARETS) {
 86:             $string = str_replace(array('<', '>'), '', $string);
 87:         }
 88: 
 89:         if ($options['remove']) {
 90:             $string = strip_tags($string);
 91:         }
 92: 
 93:         if ($options['clean_whitespace']) {
 94:             $string = preg_replace('/\s+/u', ' ', trim($string));
 95:         } else {
 96:             $string = trim($string);
 97:         }
 98: 
 99:         if ($options['encode']) {
100:             $string = htmlspecialchars($string, $options['quotes'], $options['charset']);
101:         }
102: 
103:         return $string;
104:     }

112:     public static function purify($string) {
113:         // the next few lines allow the config settings to be cached
114:         $config = HTMLPurifier_Config::createDefault();
115:         $config->set('HTML.DefinitionID', 'made by malducin');
116:         $config->set('HTML.DefinitionRev', 2);
117:         // levels describe how aggressive the Tidy module should be when cleaning up html, four levels: none, light, medium, heavy
118:         $config->set('HTML.TidyLevel', 'heavy');
119:         // check the top of your html file for the next two
120:         $config->set('HTML.Doctype', 'XHTML 1.0 Transitional');
121:         $config->set('Core.Encoding', 'UTF-8');
122: 
123:         // enable id attribute in tags
124:         $config->set('Attr.EnableID', true);
125: 
126:         // check to see if we can get raw definition and add attributes
127:         if ($def = $config->maybeGetRawHTMLDefinition()) {
128:             // adding target attribute to anchor tag
129:             $def->addAttribute('a', 'target', 'Enum#_blank,_self,_target,_top');
130:             // adding rel attibute to anchor tag
131:             $def->addAttribute('a', 'rel', 'CDATA');
132: 
133:             // Add HTML5 elements
134:             $def->addElement('figure', 'Flow', 'Flow', 'Common', array());
135:             $def->addElement('figcaption', 'Flow', 'Flow', 'Common', array());
136:         }
137: 
138:         $purifier = new HTMLPurifier($config);
139:         $cleanInput = $purifier->purify($string);
140: 
141:         return $cleanInput;
