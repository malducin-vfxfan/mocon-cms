 3:  * Custom Blowfish password hasher.
 4:  *
 5:  * Hashes passwords using Blowfish encryption. It contains several
 6:  * more features than the standard CakePHP implementation:
 7:  *
 8:  * - Can be configured via the project configuration file
 9:  *   (cost and salt prefix).
10:  * - Salt generation is more secure using mcrypt_create_iv to get
11:  *   more entropy.
12:  * - Check the returned hash lenghth.
13:  *
14:  * We need at least PHP 5.3, preferably 5.3.7 or higher.
15:  *
16:  * References:
17:  *
18:  * http://blog.ircmaxell.com/2012/12/seven-ways-to-screw-up-bcrypt.html
19:  * http://stackoverflow.com/questions/10676542/bcrypt-and-random-salts-together-in-database
20:  * http://www.php.net/manual/en/function.crypt.php
21:  *
22:  * @author        Manuel Alducin
23:  * @copyright     Copyright (c) 2009-2014, VFXfan (http://vfxfan.com)
24:  * @link          http://vfxfan.com VFXfan
25:  * @package       vfxfan-base.Controller.Component.Auth
26:  */
27: App::uses('AbstractPasswordHasher', 'Controller/Component/Auth');

29: class BlowfishAdvancedPasswordHasher extends AbstractPasswordHasher {
30: 
31: /**
32:  * One way encryption using the PHP crypt() function.
33:  *
34:  * @param string $password The string to be encrypted.
35:  * @param mixed $salt false to generate a new salt or an existing salt.
36:  * @return string The hashed string or an empty string on error.
37:  */
38:     public function hash($password = null, $salt = false) {
39:         $cost = Configure::read('Security.BlowfishAdvanced.cost');
40:         if (!$cost) $cost = 10;
41:         if ($cost < 4 || $cost > 31) {
42:             throw new Exception('Invalid value, cost must be between 4 and 31');
43:         }
44: 
45:         if ($salt === false) {
46:             $salt = self::_salt(22);
47:             $salt = vsprintf('$%s$%02d$%s', array(Configure::read('Security.BlowfishAdvanced.salt_prefix'), $cost, $salt));
48:         }
49:         if (!strlen($salt) == 29) {
50:             throw new Exception('Error generating the salt.');
51:         }
52: 
53:         $hash = crypt($password, $salt);
54:         if (strlen($hash) != 60) {
55:             throw new Exception('Error hashing the password.');
56:         }
57: 
58:         return $hash;
59:     }
60: 
61:     public function check($password = null, $hashedPassword = null) {
62:         if (!$password || !$hashedPassword) return false;
63:         else return $hashedPassword === $this->hash($password, $hashedPassword);
64:     }
65: 
66: /**
67:  * Generates a pseudo random salt suitable for use with the PHP crypt() function.
68:  * The salt length should not exceed 27. The salt will be composed of
69:  * [./0-9A-Za-z]{$length}.
70:  *
71:  * @param integer $length The length of the returned salt
72:  * @return string The generated salt
73:  */
74:     private static function _salt($length = 22) {
75:         if (defined('MCRYPT_DEV_URANDOM')) {
76:             return substr(strtr(base64_encode(mcrypt_create_iv(16, MCRYPT_DEV_URANDOM)), '+', '.'), 0, $length);
77:         } else {
78:             throw new Exception('The MCRYPT_DEV_URANDOM source is required (PHP 5.3+).');
79:         }
80:     }

82: }
