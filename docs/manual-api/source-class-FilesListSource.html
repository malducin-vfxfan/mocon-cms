 18:  * FilesListSource Data Source
 19:  *
 20:  */
 21: class FilesListSource extends DataSource {
 22: /**
 23:  * An optional description of the datasource
 24:  *
 25:  * @var string
 26:  */
 27:     public $displayField = 'A files list datasource.';

 29: /**
 30:  * Since datasources normally connect to a database there are a few things
 31:  * we must change to get them to work without a database.
 32:  */
 33: /**
 34:  * listSources() is for caching. You'll likely want to implement caching in
 35:  * your own way with a custom datasource. So just 'return null'.
 36:  */
 37:     public function listSources($data = null) {
 38:         return null;
 39:     }

 41: /**
 42:  * calculate() is for determining how we will count the records and is
 43:  * required to get update() and delete() to work.
 44:  *
 45:  * We don't count the records here but return a string to be passed to
 46:  * read() which will do the actual counting. The easiest way is to just
 47:  * return the string COUNT and check for it in read() where
 48:  * $data['fields'] == 'COUNT'.
 49:  */
 50:     public function calculate(Model $Model, $func, $params = array()) {
 51:         return 'COUNT';
 52:     }

 54: /**
 55:  * Implement the R in CRUD. Calls to Model::find() arrive here.
 56:  */
 57:     public function read(Model $Model, $queryData = array(), $recursive = null) {
 58:         // check to see if we defined a directory in the paginate conditions
 59:         if (!empty($queryData['conditions']['folder'])) {
 60:             $folder = $queryData['conditions']['folder'];
 61:         } else {
 62:             $folder = WWW_ROOT.'img';
 63:         }
 64: 
 65:         $dir = new Folder($folder);
 66:         $files = $dir->find('.*', true);
 67: 
 68:         /**
 69:          * Here we do the actual count as instructed by our calculate()
 70:          * method above. We just count the number of elements in the
 71:          * files array.
 72:          *
 73:          * We also call the funtion to sort and paginate the results.
 74:          */
 75: 
 76:         if ($files) {
 77:             if ($queryData['fields'] == 'COUNT') {
 78:                 return array(array(array('count' => count($files))));
 79:             }
 80: 
 81:             // used for sorting
 82:             $files = $this->__sortItems($files, $queryData['order'][0]);
 83: 
 84:             //used for pagination
 85:             $files = $this->__getPage($files, $queryData);
 86: 
 87:         } else {
 88:             if ($queryData['fields'] == 'COUNT') {
 89:                 return array(array(array('count' => count($files))));
 90:             }
 91:         }
 92: 
 93:         /**
 94:          * Return the results in an array which uses the model alias
 95:          * name.
 96:          */
 97:         return array($Model->alias => $files);
 98:     }

100: /**
101:  * Sort the files list
102:  *
103:  * We use the sort constants defined for standard multi sort
104:  * arrays, just for the heck of it.
105:  */
106:     private function __sortItems($files, $order) {
107:         if (empty($order)) {
108:             return $files;
109:         }
110: 
111:         switch(strtolower($order)) {
112:             case 'asc':
113:                 $direction = SORT_ASC;
114:                 break;
115:             case 'desc':
116:                 $direction = SORT_DESC;
117:                 break;
118:             default:
119:                 throw new CakeException('Invalid sorting direction');
120:         }
121: 
122:         if ($direction == SORT_ASC) {
123:             sort($files);
124:         } else {
125:             rsort($files);
126:         }
127: 
128:         return $files;
129:     }
130: 
131: /**
132:  * Get a paginated result using array slicing. We also set the
133:  * default limit to 20, like regular pagination.
134:  */
135:     private function __getPage($files = null, $data = array()) {
136:         if (empty($data['limit'])) {
137:             return $files;
138:         }
139: 
140:         if (!isset($data['limit']) || empty($data['limit'])) {
141:             $limit = 20;
142:         } else {
143:             $limit = $data['limit'];
144:         }
145:         $page = $data['page'];
146: 
147:         $offset = $limit * ($page - 1);
148: 
149:         return array_slice($files, $offset, $limit);;
150:     }
