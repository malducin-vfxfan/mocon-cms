 74:  * @param string $location
 75:  * @param string $id

 80:     public function idImage($location = null, $id = null, $options = array(), $location_default = null) {
 81:         $default = array(
 82:             'alt' => 'Image',
 83:             'class' => 'img-thumbnail',
 84:         );
 85: 
 86:         $options = array_merge($default, $options);
 87: 
 88:         if (!(isset($options['title'])) or !$options['title']) $options['title'] = $options['alt'];
 89: 
 90:         if (!$location or !$id) return;
 91: 
 92:         $dir = new Folder(WWW_ROOT.'img'.DS.$location);
 93:         $images = $dir->find(sprintf("%010d", $id).'.*', true);
 94: 
 95:         $image_name = '';
 96:         foreach ($images as $image) {
 97:             $image = new File($dir->pwd().DS.$image);
 98:             $ext = $image->ext();
 99: 
100:             if (in_array($ext, $this->exts)) {
101:                 $image_name = $image->name;
102:                 break;
103:             }
104:         }
105:         if ($image_name) {
106:             $img_link = $this->Html->image($location.'/'.$image_name, $options);
107:         } else {
108:             // if default image is in the same location
109:             if (!$location_default) {
110:                 $img_link = $this->Html->image($location.'/'.'default.png', $options);
111:             } else {
112:                 // use the specified location of the default image
113:                 // usually for images that are in subfolders like posts and events where the
114:                 // default is in the enclosing folder
115:                 $img_link = $this->Html->image($location_default.'/'.'default.png', $options);
116:             }
117:         }
