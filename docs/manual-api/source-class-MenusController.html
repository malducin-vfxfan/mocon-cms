 39: /**
 40:  * menu method
 41:  *
 42:  * Return an array of threaded menu item to be displayed (via a Helper).
 43:  * Priority must be greater than zero to display the menu item.
 44:  *
 45:  * @return array
 46:  */
 47:     public function menu() {
 48:         //  checks to make sure method is actually originating from requestAction()
 49:         if (empty($this->request->params['requested'])) {
 50:             throw new ForbiddenException();
 51:         }
 52: 
 53:         // get menu items from cache, if expired get elements and cache
 54:         $menuItems = Cache::read('menu', 'medium');
 55:         if ($menuItems === false) {
 56:             // if cache expired or non-existent, get latest
 57:             $options = array(
 58:                 'conditions' => array('Menu.priority >' => 0),
 59:                 'recursive' => -1
 60:             );
 61:             $menuItems = $this->Menu->find('threaded', $options);
 62:             Cache::write('menu', $menuItems, 'medium');
 63:         }
 64: 
 65:         return $menuItems;
 66:     }
