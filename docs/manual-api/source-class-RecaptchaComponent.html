 70:     public function initialize(Controller $controller, $settings = array()) {
 71:         if ($controller->name == 'CakeError') {
 72:             return;
 73:         }
 74:         $this->privateKey = Configure::read('Recaptcha.privateKey');
 75:         $this->Controller = $controller;
 76:

 79:         }
 80: 
 81:         $defaults = array(
 82:             'modelClass' => $this->Controller->modelClass,
 83:             'errorField' => 'recaptcha',
 84:             'actions' => array()
 85:         );
 86: 
 87:         $this->settings = array_merge($defaults, $settings);
 88:         $this->actions = array_merge($this->actions, $this->settings['actions']);
 89:         unset($this->settings['actions']);

 92:     /**
 93:      * Callback

 95:      * @param object Controller object
 96:      */
 97:     public function startup(Controller $controller) {
 98:         extract($this->settings);
 99:         if ($this->Controller->Components->enabled('Recaptcha')) {
100:             // Automatic mode .. using actions
101:             if (in_array($this->Controller->action, $this->actions)) {
102:                 $this->Controller->helpers[] = 'Recaptcha.Recaptcha';
103: 
104:                 $this->Controller->{$modelClass}->recaptcha = true;
105:                 $this->Controller->{$modelClass}->Behaviors->load('Recaptcha.Recaptcha', array(
106:                     'errorField' => $errorField
107:                 ));
108:                 $helper_options['useActions'] = true;
109:                 if (!$this->verify()) {
110:                     $this->Controller->{$modelClass}->recaptcha = false;
111:                     $this->Controller->{$modelClass}->recaptchaError = $this->error;
112:                     $helper_options['error'] = $this->error;
113:                 }
114:                 // Manual mode
115:             } else {
116:                 $helper_options['useActions'] = false;
117:                 if (!$this->verify()) {
118:                     $helper_options['error'] = $this->error;
119:                 }
120:                 
121:             }
122:             $this->Controller->helpers['Recaptcha.Recaptcha'] = $helper_options;
123:             //debug($this->Controller->helpers);
124:         }

127:     /**
128:      * Verifies the recaptcha input
129:      *
130:      * Please note that you still have to pass the result to the model and do
131:      * the validation there to make sure the data is not saved!
132:      *
133:      * @return boolean True if the response was correct
134:      */
135:     public function verify() {
136:         if (isset($this->Controller->request->data['recaptcha_challenge_field']) &&
137:                 isset($this->Controller->request->data['recaptcha_response_field'])) {
138: 
139:             $response = $this->_getApiResponse();
140:             $response = explode("\n", $response);
141: 
142:             if (empty($response[0])) {
143:                 $this->error = __d('recaptcha', 'Invalid API response, please contact the site admin.', true);
144:                 return false;
145:             }
146: 
147:             if ($response[0] == 'true') {
148:                 return true;
149:             }
150: 
151:             if ($response[1] == 'incorrect-captcha-sol') {
152:                 $this->error = __d('recaptcha', 'Incorrect captcha', true);
153:             } else {
154:                 $this->error = $response[1];
155:             }
156: 
157:             return false;
158:         }
159:     }

161:     /**
162:      * Queries the Recaptcha API and and returns the raw response
163:      *
164:      * @return string
165:      */
166:     protected function _getApiResponse() {
167:         $Socket = new HttpSocket();
168:         return $Socket->post($this->apiUrl, array(
169:                     'privatekey' => $this->privateKey,
170:                     'remoteip' => env('REMOTE_ADDR'),
171:                     'challenge' => $this->Controller->request->data['recaptcha_challenge_field'],
172:                     'response' => $this->Controller->request->data['recaptcha_response_field']));
173:     }
